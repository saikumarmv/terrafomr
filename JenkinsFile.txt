pipeline {
    agent any
    tools {
        terraform 'terraform'
    }

    stages {
        stage('Checkout') {
            steps {
                checkout([$class: 'GitSCM', branches: [[name: '*/master']], extensions :[], userRemoteConfig: [[url: 'https://github.com/saikumarmv/terrafomr/tree/master']]])
            }
        }
            stage('terraform init') {
            steps {
                bat 'terraform init'
                bat 'terraform version'
            }
            }
        stage('terraform Action') {
            steps {
            withCredentials([string(credentialsId: 'access_key', variable: 'access_key'), string(credentialsId: 'secret_key', variable: 'secret_key')]) {
                echo "Terraform action is --> ${action}"
                export TF_VAR_access_key='${access_key}'
                export TF_VAR_secret_key='${secret_key}'
                bat "terraform ${action} --auto-approve"
}
            }
        }
    }
}
